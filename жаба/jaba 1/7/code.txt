// 1. Абстрактний базовий клас
abstract class Employee {
    private String name;

    public Employee(String name) {
        this.name = name;
    }

    public String getName() {
        return name;
    }

    // Абстрактний метод (віртуальна функція), який ОБОВ'ЯЗКОВО 
    // має бути реалізований у всіх похідних класах
    public abstract double calculateSalary();

    @Override
    public String toString() {
        return "Працівник: " + name + ", Нараховано: " + String.format("%.2f", calculateSalary()) + " грн";
    }
}

// 2. Працівник з погодинною оплатою
class HourlyEmployee extends Employee {
    private double hourlyRate;
    private double hoursWorked;

    public HourlyEmployee(String name, double hourlyRate, double hoursWorked) {
        super(name);
        this.hourlyRate = hourlyRate;
        this.hoursWorked = hoursWorked;
    }

    @Override
    public double calculateSalary() {
        return hourlyRate * hoursWorked;
    }
}

// 3. Штатний працівник (фіксована ставка)
class SalariedEmployee extends Employee {
    private double monthlySalary;

    public SalariedEmployee(String name, double monthlySalary) {
        super(name);
        this.monthlySalary = monthlySalary;
    }

    @Override
    public double calculateSalary() {
        return monthlySalary;
    }
}

// 4. Працівник з процентною ставкою (від продажів)
class CommissionEmployee extends Employee {
    private double salesAmount;
    private double commissionPercentage;

    public CommissionEmployee(String name, double salesAmount, double commissionPercentage) {
        super(name);
        this.salesAmount = salesAmount;
        this.commissionPercentage = commissionPercentage;
    }

    @Override
    public double calculateSalary() {
        return salesAmount * (commissionPercentage / 100);
    }
}

// Головний клас для демонстрації
public class Main {
    public static void main(String[] args) {
        // Створюємо масив абстрактних працівників, 
        // що містить об'єкти різних конкретних класів
        Employee[] employees = new Employee[3];

        employees[0] = new HourlyEmployee("Олександр (Погодинна)", 150.0, 160);
        employees[1] = new SalariedEmployee("Марія (Штат)", 25000.0);
        employees[2] = new CommissionEmployee("Дмитро (Відсоток)", 500000.0, 5.0);

        System.out.println("--- Відомість нарахування заробітної плати ---");
        
        // Демонстрація поліморфізму: для кожного об'єкта 
        // викликається СВІЙ метод calculateSalary()
        for (Employee emp : employees) {
            System.out.println(emp.toString());
        }
    }
}
